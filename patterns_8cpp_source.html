664class Synchronized_queue
665 : Container
666{
667 mutex mtx;
668public:
669 condition_variable cv;
670 bool stop = false;
671 void push(T&& v) {
672 scoped_lock<mutex>{mtx}, Container::push(v);
673 cv.notify_one();
674 };
675
676 optional<reference_wrapper<T>> pull() {
677 unique_lock<mutex> lk(mtx);
678 cv.wait(lk, [&] { return !this->empty() || stop; });
679 optional<reference_wrapper<T>> ret;
680 if (stop) {
681 ret = nullopt;
682 } else {
683 ret = make_optional(ref(Container::front()));
684 this->pop();
685 }
686 return ret;
