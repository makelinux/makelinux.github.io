337static void lambda_basics(void)
338{
339 auto annotated_named_lambda_expression = // optional name
340 [ ] // capture clause
341 ( ) // optional list of arguments
342 { }; // body
343
344 // Primitive named lambdas are just like closure functions:
345 // https://en.wikipedia.org/wiki/Closure_(computer_programming)
346
347 // declaration like a function:
348 // void closure() { };
349 auto closure = [] { };
350
351 closure();
352
353 // with arguments
354 auto pass = [] (int a) { return a; };
355 assert(pass(5) == 5);
356
357 // lambda captures external value
358 int c = 1;
359 auto get_i = [=] () { return c; };
360 assert(get_i() == 1);
361
362 // lambda captures external variable by reference
363 // with omitted arguments and return type
364 auto inc_get = [&] { return ++c; };
365 assert(inc_get() == 2);
366 assert(inc_get() == 3);
367
368 // annotated expanded empty inline lambda call:
369 [ ] // capture
370 ( ) // optional list of arguments
371 -> void // optional return value
372 { } // body
373 ( ); // call with arguments
374
375 // annotated expanded sample inline lambda call:
376 c = // result
377 [c] // capture
378 (int a) // an argument
379 -> int // return value
380 { return c + a; } // body
381 (1); // call with argument
382 assert(c == 4);
383
384 // inline lambda which is called in place
385 // https://en.wikipedia.org/wiki/Anonymous_function
386
387 // assert((1 + 1) == 2);
388 assert([](int a) { return a + 1; }(1) == 2);
389
390 // Actually calling lambda inline is useless
391 // and is provided only for demonstration.
392}
